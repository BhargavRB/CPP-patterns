#include <iostream>
using namespace std;

void pattern1(int n)
{
    int i,j;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++){
            cout << '*';
        }
        cout << endl;
    }
}

int main() {
    int n;
    cout << "Enter Number :- ";
    cin >> n;
    pattern1(n);
    return 0;
}
